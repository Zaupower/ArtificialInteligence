/*REGRAS DE BUSCA*/
/*faz a busca dos pokemons por determinado tipo*/
tipo_pokemon(X,Y):-pokemon(X,tipo(Y)).

/*O pokemon X é híbrido qundo pertence ao tipo Y e Z ao mesmo tempo*/
tipo_hibrido(X, Y, Z):-(pokemon(X, (tipo(Y))), pokemon(X, tipo(Z)), Y\==Z).



/*OFENSIVA aqui vai um tipo(X) e um tipo(Y)*/
/*O tipo X tem desvantagem ofensiva para o tipo Y se o tipo Y tem vantagem defensiva para o tipo X*/
desvantagem_ofensiva(X,Y):-vantagem_defensiva(tipo(Y), tipo(X)).


/*##############################################################################################################################*/
/*DEFENSIVA aqui vai um tipo(X) e um tipo(Y)*/

/*O tipo X tem desvantagem defensiva para o tipo Y se o tipo Y tem vantagem ofensiva para o tipo X*/
desvantagem_defensiva(X,Y):-vantagem_ofensiva(tipo(Y), tipo(X)).


/*@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@*/

/*OFENSIVA aqui vai um pokemon(X) e um tipo(Z)*/
/*O pokemon X tem o dobro de ataque para pokemons do tipo Z se o pokemon X for do tipo Y*/
/*pokemon_vantagem_ofensiva_tipo(X,Y):-pokemon(X, tipo(Z)), vantagem_ofensiva(tipo(Z), tipo(Y)).*/

pokemon_vantagem_ofensiva_tipo(X,Y):-(nome(X),(
	tipo_hibrido(X,W,Z)->(/*caso for do tipo "híbrido"*/
		tipo(Y),(
			vantagem_ofensiva(tipo(Z), tipo(Y))->(/*O pokemon X tem vantagem para o tipo Y se um de seus tipos for Z*/
				vantagem_ofensiva(tipo(Z), tipo(Y))
			)
			;(/*Caso um dos tipos não for Z, caso for não passa por aqui, pois não importa o seu outro tipo, o pokemon já tem vantagem para o tipo Y */
				vantagem_ofensiva(tipo(W), tipo(Y))->(/*O pokemon X tem vantagem para o tipo Y se um de seus tipos for W*/
					vantagem_ofensiva(tipo(W), tipo(Y))
				)
			)
		)
		
	)
	;(/*caso nao for do tipo "híbrido"*/
		pokemon(X, tipo(Z)), vantagem_ofensiva(tipo(Z), tipo(Y)) /*O pokemon X tem vantagem p/ o tipo Y se for do tipo Z*/
	)
)).


/*@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@*/

/*O pokemon X tem metado do ataque para pokemons tipo Z se o pokemon X for do tipo Y*/
/*pokemon_desvantagem_ofensiva_tipo(X,Z):-pokemon(X, tipo(Y)), desvantagem_ofensiva(Y,Z).*/

pokemon_desvantagem_ofensiva_tipo(X,Y):-(nome(X),(
	tipo_hibrido(X,W,Z)->(/*caso for do tipo hibrido*/
		tipo(Y),(
			desvantagem_ofensiva(Z,Y)->(/*O pokemon X tem desvantagem para o tipo Y se um de seus tipos for Z*/
				desvantagem_ofensiva(Z,Y)
			)
			;(/*Caso um dos tipos não for Z, caso for não passa por aqui, pois não importa o seu outro tipo, o pokemon já tem desvantagem para o tipo Y */
				desvantagem_ofensiva(W,Y)->(/*O pokemon X tem desvantagem para o tipo Y se um de seus tipos não for W*/
					desvantagem_ofensiva(W,Y)
				)				
			)

		)

	)
	;(/*caso não for do tipo híbrido*/
		pokemon(X, tipo(Z)), desvantagem_ofensiva(Z,Y)/*O pokemon X tem desvantagem p/ o tipo Y se for do tipo Z*/

	)
)).


/*@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@@*/

/*O pokemon X nao é efetivos em ataques a pokemons do tipo Z quando o pokemon X for do tipo Y*/
/*pokemon_nao_ataca_tipo(X,Z):-pokemon(X,tipo(Y)), sem_ataque(tipo(Y), tipo(Z)).*/

pokemon_nao_ataca_tipo(X,Y):-(nome(X),(
	tipo_hibrido(X,W,Z)->(/*caso for do tipo hibrido*/
		tipo(Y),(
			sem_ataque(tipo(Z), tipo(Y)), sem_ataque(tipo(W), tipo(Y))/*O pokemon X não tem efetividade para o tipo Y se os dois tipos forem Z e W*/			
		)

	)
	;(/*Caso não for do tipo hibrido*/
		pokemon(X,tipo(Z)), sem_ataque(tipo(Z), tipo(Y))
	)


)).
